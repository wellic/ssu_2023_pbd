pwd
cd lab
cd ssu_2023_pbd

# Підтягнемо оновлення з основного репозиторію (UPSTREAM)
git remote add upstream https://github.com/wellic/ssu_2023_pbd
git fetch -pt upstream
git checkout master
git rebase upstream/master
git push -fu origin master:master
